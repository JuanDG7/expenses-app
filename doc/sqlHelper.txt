*************4 TIPOS DE JOINS*************

-- INNER JOIN (o simplemente JOIN)
-- Devuelve solo las filas que tienen coincidencias en ambas tablas.
SELECT * FROM table1 INNER JOIN table2 ON table1.id = table2.id;

-- LEFT JOIN (o LEFT OUTER JOIN)
-- Devuelve todas las filas de la tabla izquierda y las filas coincidentes de la tabla derecha. 
-- Si no hay coincidencia, se devuelve NULL en la tabla de la derecha.
SELECT * FROM table1 LEFT JOIN table2 ON table1.id = table2.id;

-- RIGHT JOIN (o RIGHT OUTER JOIN)
-- Devuelve todas las filas de la tabla derecha y las filas coincidentes de la tabla izquierda. 
-- Si no hay coincidencia, se devuelve NULL en la tabla de la izquierda.
SELECT * FROM table1 RIGHT JOIN table2 ON table1.id = table2.id;

-- FULL JOIN (o FULL OUTER JOIN)
-- Devuelve todas las filas cuando hay una coincidencia en cualquiera de las tablas. 
-- Si no hay coincidencia, se devuelve NULL en la tabla que no tiene la coincidencia.
SELECT * FROM table1 FULL JOIN table2 ON table1.id = table2.id;

*************AGREGGATE FUNCTIONS*************

COUNT(): Devuelve el número de valores en un grupo de valores.
Uso: COUNT(column_name)

SUM(): Devuelve la suma de un grupo de números.
Uso: SUM(column_name)

AVG(): Devuelve el promedio de un grupo de números.
Uso: AVG(column_name)

MIN(): Devuelve el valor mínimo de un grupo de números.
Uso: MIN(column_name)

MAX(): Devuelve el valor máximo de un grupo de números.
Uso: MAX(column_name)


/Operan sobre columnas completas

/No puedes mezclar columnas normales con funciones agregadas sin GROUP BY      <----
SELECT id, MAX(id) FROM comments;                                           <--- id AQUI ESTA MAL

/Cuando usas una función agregada, el resultado es siempre 1 solo valor
SELECT MAX(id) FROM comments;

/Se usan mucho junto con GROUP BY
Porque GROUP BY divide los datos en grupos, y el agregado calcula sobre cada grupo.
SELECT user_id, COUNT(*) 
FROM comments 
GROUP BY user_id;


/No te olvides del orden de ejecución

WHERE → filtra antes del agregado
GROUP BY → agrupa
HAVING → filtra después del agregado
SELECT → muestra resultados

/Agregados ignoran NULL (excepto COUNT(*))
Ejemplo: si hay valores NULL en price:

SUM(price) no suma los NULL
AVG(price) tampoco los considera
COUNT(price) no los cuenta
pero COUNT(*) sí cuenta la fila aunque el valor sea NULL

WITH RECURSIVE suggestions(leader_id, follower_id, depth) AS (
		SELECT leader_id, follower_id, 1 AS depth
		FROM followers
		WHERE follower_id = 1000
  	UNION
	    SELECT followers.leader_id, followers.follower_id, depth+1
		FROM followers
		JOIN suggestions ON suggestions.leader_id = followers.follower_id
		WHERE depth < 3
)
SELECT DISTINCT users.id, users.username
FROM suggestions
JOIN users ON users.id = suggestions.leader_id
WHERE depth > 1
LIMIT 30;

************date_trunc**************************************************
*redondea dia, semana, anho       


SELECT 
    date_trunc('week', COALESCE(posts.created_at, comments.created_at)) AS week,      <--COALESCE devuelve el valor TRUE
    COUNT(posts.id) AS num_likes_for_posts,                             <--- este 'count' NO CUENTA los valores NULL que encuentre    
    COUNT(comments.id) AS num_likes_for_comments
FROM likes
LEFT JOIN posts ON posts.id=likes.post_id
LEFT JOIN comments ON comments.id=likes.comment_id
GROUP BY week
ORDER BY week 

**********MATERIALIZED VIEW****************************************************
*Es como el memo de javascript, guarda el resultado para no volver a ejecutar el codigo ya que es una operacion costosa
*si ocurriria alguna modificacion en alguna de las tablas, habria que actualizar esto con REFRESH MATERIALIZED VIEW para no tener
datos desactualizados


CREATE MATERIALIZED VIEW weekly_likes AS (
SELECT 
    date_trunc('week', COALESCE(posts.created_at, comments.created_at)) AS week,      <--COALESCE devuelve el valor TRUE
    COUNT(posts.id) AS num_likes_for_posts,                             <--- este 'count' NO CUENTA los valores NULL que encuentre    
    COUNT(comments.id) AS num_likes_for_comments
FROM likes
LEFT JOIN posts ON posts.id=likes.post_id
LEFT JOIN comments ON comments.id=likes.comment_id
GROUP BY week
ORDER BY week                 <-------detalle, aqui no va ningun punto y coma ;
) WITH DATA;                   <-- WITH DATA, va a guardar el resultado de la operacion  <--- recien aqui va el ;

********REFRESH MATERIALIZED VIEW weekly_likes******************************************************
*Sirve para actualizar lo que ya se guardo en materialized view
REFRESH MATERIALIZED VIEWS weekly_likes;
